\subsection{Замечательные пределы}

\begin{definition}
    Неопределенностью вида $\frac{0}{0} \l(\frac{\infty }{\infty  }\r)$ называется:
    \[
        \Lim{x}{x_0} \frac{f(x)}{g(x)},\ \Lim{x}{x_0} f(x) = \Lim{x}{x_0} g(x) = 0\ (\infty ),\ x_0 \in \overline{\R} = \R \cup \{\pm \infty \}
    \]
\end{definition}

Неопределенность --- т.к ответ может быть любым

\begin{exercise}
    $\cos x$ непрерывен на $\R$
\end{exercise}

\begin{solution}
    \[
        \Lim{x}{x_0} \cos x = \cos x_0
    \]
    \[
        \Updownarrow
    \]
    \[
        \cos x - \cos x_0 = \overline{o}(1) \text{ при }  x \to x_0
    \]
    \[
        \l| -2 \cdot \sin \frac{x + x_0}{2} \cdot \sin \frac{x - x_0}{2} \r| \leq 2 \cdot 1 \cdot \l|\frac{x - x_0}{2}\r| \to 0
    \]
\end{solution}

\subsubsection{Первый замечательный предел}

\begin{theorem}[Первый замечательный предел]
    $$\Lim{x}{0} \frac{\sin x}{x} = 1$$
\end{theorem}

\begin{figure}[h]
  \centering
  \includegraphics[width=0.3\textwidth]{lectures/files/lec_10_14-13-30.png}
  \label{fig:lec_10_14-13-30.png}
\end{figure}

\begin{Proof}
Рассмотрим единичную окружность ($R=1$) и угол $x \in (0, \pi/2)$.
Сравним площади:
1. Площадь внутреннего треугольника: $S_{\triangle} = \frac{1}{2} R^2 \sin x = \frac{\sin x}{2}$.

2. Площадь сектора: $S_{sect} = \pi \cdot \frac{x}{2\pi } = \frac{x}{2}$.

3. Площадь внешнего треугольника (основание 1, высота $\tan x$): $S_{\text{внеш}} = \frac{1}{2} \tan x$.

Из чертежа очевидно: $S_{\triangle} < S_{sect} < S_{\text{внеш}}$.
Отсюда: $\frac{\sin x}{2} < \frac{x}{2} < \frac{\tan x}{2} \implies \sin x < x < \tan x$.
\end{Proof}
\begin{Proof}[]
Как мы знаем, в силу нечетности $\sin x : \frac{\sin x}{x} = \frac{\sin |x|}{|x|}$ и четности $\cos x : \cos |x| = \cos x$

Рассмотрим проколотую окрестность $\overset{\circ}{U}_{\pi/2}(0)$. 
Для $x \in (0, \pi/2)$ из геометрических соображений (сравнение площадей сектора и треугольников) справедливо:
\[ \sin x < x < \tan x. \]
Разделим все части на $\sin x > 0$:
\[ 1 < \frac{x}{\sin x} < \frac{1}{\cos x} \implies \cos x < \frac{\sin x}{x} < 1. \]
Так как функции $\cos x$ и $f(x) = \frac{\sin x}{x}$ являются четными, данное двойное неравенство верно для любого $x \in \overset{\circ}{U}_{\pi/2}(0)$ (включая отрицательные значения).

Поскольку $\lim_{x \to 0} \cos x = 1$ и $\lim_{x \to 0} 1 = 1$, то по теореме о зажатой функции:
\[ \lim_{x \to 0} \frac{\sin x}{x} = 1. \]
\end{Proof}

\subsubsection{Второй замечательный предел}

\begin{theorem}[Второй замечательный предел]
\[
    \Lim{x}{0} (1 + x)^\frac{1}{x} = e
\]
\end{theorem}

\begin{Proof}
    Мы знаем:

    $$\Lim{n}{\infty} \l(1 + \frac{1}{n}\r)^n = e$$

    Хотим:
    
    \[
    \Lim{t}{+\infty} \l(1 + \frac{1}{t}\r)^t = e \tag{1}
    \]

    Воспользуемся неубываением $y = x^t$
    \[
        \underbrace{\l(1 + \frac{1}{[t] + 1} \r)^{[t]}}_{g(t)} \leq \underbrace{\l(1 + \frac{1}{t} \r)^{t}}_{f(t)} \leq \underbrace{\l(1 + \frac{1}{[t]} \r)^{[t] + 1}}_{h(t)},\ \fbox{$t \geq 1, t \in \R$}
    \]

    Мы знаем: $\Lim{n}{\infty} \l(1 + \frac{1}{n + 1} \r)^{n} = e$

    \[
        \forall \eps > 0\ \exists N(\eps ) \in \N\ \forall n > N(\eps )\ \l|\l(1 + \frac{1}{n + 1}\r)^n - e \r| < \eps 
    \]

    {\bf Утверждение:} $\Lim{t}{+\infty} g(t) = e$

    \[
        \forall \eps > 0\ \exists M(\eps )\ \forall t > M(\eps ) \l|\l(1 + \frac{1}{[t] + 1} \r)^{[t]} - e\r| < \eps
    \]

    $M(\eps) = N(\eps) + 1$, так как:
    \[
        \forall t > N(\eps ) + 1 : [t] > N(\eps )
    \]

    Аналогично доказывается $h(t) \xrightarrow[t \to \infty ]{} e$

    По теореме о зажатой функции:
    \[
        g(t) \to e, h(t) \to e \Rightarrow f(t) \to e
    \]
\end{Proof}

{\bf Следствие 1} 
\[
    \Lim{x}{0} \frac{\ln (1 + x)}{x} = 1
\]
\[
    \frac{1}{x} \cdot \ln (1 + x) = \ln (1 + x)^{\frac{1}{x}} \rightarrow \ln e = 1

    \Lim{x}{x_0} f(x) = f(x_0) = f(\Lim{x}{x_0} x)

\subsection{Непрерывность функции на отрезке}

\begin{definition}
    Функция непрерывна на $[a; b]$, если

    \begin{itemize}
        \item[1)] $[a;b] \subset D_f$
        \item[2)] $f(x)$ непрерывна $ \forall x_0 \in (a;b)$
        \item[3)] $\Lim{x}{a^+} f(x) = f(a)\ \Lim{x}{b^-} f(x) = f(b)$
    \end{itemize}
\end{definition}

\begin{remark}
    В дальнейшем будем считать что $D_f = [a;b]$
\end{remark}

\begin{theorem}
    Если функция непрерывна на отрезке $[a;b]$, то 
    \begin{itemize}
        \item[1)] она ограничена на $[a;b]$
        \item[2)] достигает $\sup f(x)\ x \in [a;b] = \sup E_f$ и $\inf f(x)\ x \in [a;b] = \inf E_f$
    \end{itemize}
\end{theorem}

\begin{Proof}
    Пусть $M = \sup f(x)\ x \in [a;b] \in \overline{\R}, m = \inf f(x)\ x \in [a;b] \in \overline{\R}$

    Хотим (докажем сверху):
    \begin{itemize}
        \item[1)] $M \in \R$
        \item[2)] $\exists x_0 : f(x_0) = M$ 
    \end{itemize}

    Докажем 2), 1) следует из него (очевидно, что если $\exists x_0 : f(x_0) = M$, то $x \neq \infty$)

    Построим последовательность $a_n \uparrow M : a_n$ строго возрастает и $\Lim{n}{\infty} a_n = M$. Тогда: 
    \[
        \forall n\ \exists x_n \in [a;b]\ f(x_n) \geq a_n
    \]
\end{Proof}

\begin{figure}[h]
  \centering
  \includegraphics[width=0.3\textwidth]{lectures/files/2239703.png}
  \label{fig:2239703.png}
\end{figure}

\begin{Proof}[]
Предположим противное:
\[
    \exists n_0\ \forall x \in [a;b]\ f(x) < a_{n_0}
\]
$a_{n_0}$ --- верхняя грань для $E_f$, $\bot$

Построили $\{x_n\}$: 
\[
    \underbrace{a_n}_{\to M} \leq f(x_n) \leq \underbrace{M}_{\to M}
\]
\[
    \Lim{n}{\infty} f(x_n) = M
\]

$x_n \in [a;b] \Rightarrow \exists x_{n_k} : x_{n_k} \underset{k \to \infty}{\rightarrow} x_0$. Тогда по теореме Б-В она ограничена.

\[
    f(x_{n_k}) \xrightarrow[k \to \infty ]{} M
\]
Определение непрерывности по Гейне:
\[
    f(x_{n_k}) \xrightarrow[k \to \infty ]{} f(x_0)
\]
По теореме о единственности предела:
\[
    M = f(x_0)
\]
\end{Proof}

\begin{theorem}
    Непрерывная на отрезке функция принимает все промежуточные значения, т.е $f(x), [a;b]$

    $$A = f(x_1),\ B = f(x_2),\ A < B,\ x_1, x_2 \in [a;b]$$

    тогда:
    $$\forall C \in (A;B)\ \exists x_0 \in [a;b] : f(x_0) = C$$
\end{theorem}

{\bf Следствие} $f(x)\ \text{непрерывна на}\ [a;b] = D_f, \text{то}\ E_f = [m; M]$

\begin{Proof}
Построим последовательность вложенных отрезков $\l\{[a_k; b_k] \r\}_{k \in \N}$, будем считать $x_1 < x_2$

$[a_1; b_1] = [x_1; x_2]$

\[
    f(a_1) = f(x_1) < C,\ f(b_1) = f(x_2) > C
\]

Возьмем середину $x_3 = \frac{x1 + x2}{2}$

\begin{itemize}
    \item[1.] $f(x_3) = C$ ч.т.д.
    \item[2.] $f(x_3) < C$ $[a_2; b_2] = [x_3; b_1]$
    \item[3.] $f(x_3) > C$ $[a_2; b_2] = [a_1; x_3]$
\end{itemize}

По теореме Вейерштрасса:

$\{a_k\}_k$ --- неубывающая и огр сверху b $\Rightarrow a_k \xrightarrow[k \to \infty ]{} \alpha $

$\{b_k\}_k$ --- невозрастает и огр снизу a $\Rightarrow b_k \xrightarrow[k \to \infty ]{} \beta $

$\underbrace{b_k}_{\to \beta } - \underbrace{a_k}_{\to \alpha } = \frac{1}{2^{k - 1}} (x_2 - x_1) \underset{k \to \infty}{\to} 0 \Rightarrow \alpha = \beta $
\[
    \underbrace{f(a_k)}_{\xrightarrow[k \to \infty ]{} f(\alpha) } < C
\]
По теореме о предельном переходе:
\[
    f(\alpha) \leq C
\]
Аналогично $f(b_k)$

Тогда если $\alpha = \beta = x_0$:
\[
    f(x_0) = f(\alpha) = f(\beta) = C
\]
\end{Proof}
