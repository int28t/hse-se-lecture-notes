\section{Свойства определенного интеграла. Интеграл с переменным верхним пределом. Его свойства. Формула Ньютона-Лейбница.}

\subsection{Свойства определенного интеграла}
\begin{theorem}[Замечание]
    
    Если\\
    \[
    \begin{aligned}
        f(x) \in R[a; b]
    \end{aligned}
    \]
    То\\
   \[
        \forall \tau_n : d(\tau_n) \tounder{n \to \infty} 0,\quad \forall \xi_n:
        \quad\sigma(\tau_n; \xi_n) \tounder{n \to \infty} \Int{a}{b} f(x) \diff x = I
    \]
\end{theorem}
\begin{Proof}
    Есть:\\
    \[
        \forall \eps > 0~ \exists\delta(\eps)~ \forall\tau: ~d(\tau) < \delta(\eps),\quad
        \forall \xi:\quad |\sigma(\tau; \xi) - I| < \eps
    \]
    \[
        \forall\gamma > 0~\exists N(\gamma)~ \forall n > N(\gamma):~ |\sigma(\tau_n; \xi_n) - I| < \eps
    \]
    Нужно:\\
    \[
        \forall \eps > 0~ \exists N'(\eps)~ \forall n > N'(\eps):~ |\sigma(\tau_n; \xi_n - I)| < \eps\\
    \]

    Начиная с $N'(\eps) = N(\delta(\eps))$ всё выполнено
\end{Proof}

\begin{definition}
    $\boxed{1}$(Линейность)\\
    Если
    \[
        f(x), g(x) \in R[a; b]
    \]
    То
    \[
        \forall\alpha, \beta \in \R: \alpha f(x) + \beta g(x) \in R[a; b]
    \]
    И
    \[
        \Int{a}{b}{(\alpha f(x) + \beta g(x))\diff x} = \alpha\Int{a}{b}{f(x)\diff x} + \beta\Int{a}{b}{g(x)\diff x}
    \]
\end{definition}

\begin{Proof}[Докажем интегрируемость]
        Проверим критерий Лебега интегрируемости по Риману\\
        1) ограниченность очевидна\\
        2) По свойствам множеств меры ноль:\\

\begin{Proof}[Докажем связь интегралов]
    Возьмем
    \[
    \begin{aligned}
        &\tau_n: d(\tau_n) \to 0 \text{ и } \xi_n\\
        &\sigma_{\alpha f(x) + \beta g(x)}(\tau_n; \xi_n) = \Sum{k = 1}{n} \alpha f(\xi_k) + \beta g(\xi_k)|\D x_k| =  \alpha \sigma_f(\tau_n; \xi_n) + \beta \sigma_g(\tau_n; \xi_n)\\
    \end{aligned}
    \]
    Из замечания получаем
    \[
    \begin{aligned}
        &\sigma_{\alpha f(x) + \beta g(x)}(\tau_n; \xi_n) \tounder{n \to \infty}\Int{a}{b}(\alpha f(x) + \beta g(x))\diff x\\
        &\alpha \sigma_f(\tau_n; \xi_n) + \beta \sigma_g(\tau_n; \xi_n) \tounder{n \to \infty} \alpha\Int{a}{b}{f(x)\diff x} + \beta\Int{a}{b}{g(x)\diff x}\\
    \end{aligned}
    \]
    Из единственности предела последовательности
    \[
    \begin{aligned}
        \Int{a}{b}(\alpha f(x) + \beta g(x))\diff x =  \alpha\Int{a}{b}{f(x)\diff x} + \beta\Int{a}{b}{g(x)\diff x}\\
    \end{aligned}
    \]

\end{Proof}
    
\begin{definition}
    $\boxed{2}$(Аддитивность)\\
    Если\\
    \[
        f(x) \in R[a; b]
    \]
    То\\
    \[
        \forall c \in (a; b): f(x) \in R[a; c] \text{ и } f(x) \in R[c; b]
    \]
    И\\
    \[
        \Int{a}{b}{f(x)\diff x} = \Int{a}{c}{f(x)\diff x} + \Int{c}{b}{f(x)\diff x}
    \]
\end{definition}
\begin{Proof}
    1) Доказательство интегрируемости очевидно\\
    2) Связь интегралов

    Построим последовательность $\tau'_n \text{ отрезка } [a; c] \text{ и возьмем } \xi'_n$\\
    Построим последовательность $\tau''_n \text{ отрезка } [c; b] \text{ и возьмем } \xi''_n$\\
    Объединим разбиения и получим $\tau_n \text{ отрезка } [a; b] \text{ и разбиение } \xi_n$\\
    \[
        \sigma_f(\tau_n; \xi_n) = \sigma_f(\tau'_n; \xi'_n) + \sigma_f(\tau''_n; \xi''_n)
    \]
    А значит
    \[
        \Int{a}{b}{f(x)\diff x} = \Int{a}{c}{f(x)\diff x} + \Int{c}{b}{f(x)\diff x}
    \]

\end{Proof}

\begin{definition}
    $\boxed{3}$\\
    3.1) Если $f(x) \in R[a; b]$, то $|f(x)| \in R[a; b]$\\
    3.2) Если $f(x), g(x) \in R[a; b]$ и $f(x) \leq g(x) \text{ на } [a; b]$, то $\Int{a}{b}{f(x)\diff x} \leq \Int{a}{b}{g(x)\diff x}$\\
    3.3) Если $f(x) \in R[a; b]$, то $|\Int{a}{b}{f(x)\diff x}| \leq \Int{a}{b}{|f(x)|\diff x}$ - Обобщенное неравенство треугольника\\
\end{definition}

\begin{Proof}[Докажем 3.3]
    \[
    \begin{aligned}
        &-|f(x)| \leq f(x) \leq |f(x)|\\
        &-\Int{a}{b}{|f(x)|\diff x} \leq \Int{a}{b}{f(x)\diff x} \leq \Int{a}{b}{|f(x)|\diff x}\\
    \end{aligned}
    \]
\end{Proof}

\begin{theorem}[$\boxed{4}$ Теорема о среднем]
    Если $f(x)$ - непрерывна на [a; b], то:\\
    \[
        \exists\xi\in(a; b):\quad \Int{a}{b}{f(x)\diff x} = f(\xi)(b - a)
    \]
\end{theorem}

\begin{Proof}
    Обозначим:\\
    \[
    \begin{aligned}
        &m = \inf f(x)\\
        &M = \sup f(x)\\
    \end{aligned} 
    \]
    Начало доказательства:\\
    \[
    \begin{aligned}
        m &\leq f(x) \leq M\\
        m(b - a) &\leq \Int{a}{b}{f(x)\diff x} \leq M(b - a)\\
        m &\leq \frac{1}{(b - a)}\Int{a}{b}{f(x)\diff x} \leq M\\
    \end{aligned}
    \]
    \[
    \begin{aligned}
        \exists\xi \in(a; b):\quad f(\xi) = \frac{1}{(b - a)}\Int{a}{b}{f(x)\diff x}
    \end{aligned}
    \]
\end{Proof}

\begin{remark}
    В предыдущем доказательстве равенство в неравенстве возможно только в случае, если $f(x) - const$
\end{remark}


\subsection{Интеграл с переменным верхним пределом}
    \begin{definition}
        Пусть $f(x) \in R[\alpha; \beta],\quad a, x \in [\alpha; \beta]$\\
        Интегралом с переменным верхним пределом определим функцию $F(x) = \Int{a}{x}f(t)\diff t$\\
    \end{definition}

    \begin{remark}
        \[
            F(a) = 0\\
        \]
    \end{remark}


\subsection{Свойства интеграла с переменным верхним пределом}
    \begin{theorem}[Свойство $boxed{1} - $ непрерывность]
        $F(x)$ непрерывна на $[\alpha; \beta]$\\
    \end{theorem}

    \begin{Proof}
        \[
        \begin{aligned}
            x_0 \in [\alpha; \beta]\\
        \end{aligned}
        \]
        Рассмотрим $\Lim{x}{x_0}$
        \[
        \begin{aligned}
            |F(x_0 + \D x) - F(x_0)| &= \\
            |\Int{a}{x + \D x_0}f(t)\diff t - \Int{a}{x_0}f(t)\diff t| &= |\Int{x_0}{x + \D x_0}f(t)\diff t|\\
        \end{aligned}
        \]
        \[
        \begin{aligned}
            |\Int{x_0}{x + \D x_0}f(t)\diff t| \leq |\Int{x_0}{x + \D x_0}|f(t)|\diff t| \leq M |\Int{x_0}{x + \D x_0}1\diff t| = M|\D x| \tounder{\D x \to 0} 0
        \end{aligned}
        \]
    \end{Proof}

    \begin{theorem}
        Если $f(x)$ - непрерывна на $(\alpha; \beta)$, то $F(x)$ дифференцируема на $(\alpha; \beta)$ и $F'(x) = f(x)$\\
    \end{theorem}
    \begin{Proof}
        $x_0 \in (\alpha; \beta)$
        \[
        \begin{aligned}
            &\frac{F(x) - F(x_0)}{x - x_0} =\\
            &=\frac{1}{x - x_0}\Int{x_0}{x}f(x)\diff x &&=\\
            &=\frac{1}{\D x}\Int{x_0}{x_0 + \D x}f(x)\diff x &&=\\
        \end{aligned}
        \]
        По теореме о среднем:\\
        \[
        \begin{aligned}
            \exists &\xi \in (x_0; x_0 + \D x): \qquad
            f(\xi) \tounder{\D x \to 0} f(x_0)\\
            &\xi \tounder{\D x \to 0} x_0   
        \end{aligned}
        \]
        А значит
        \[
        \begin{aligned}
            &\frac{F(x) - F(x_0)}{x - x_0} \tounder{x\to x_0} f(x_0)\\
            &\Lim{x}{x_0}\frac{F(x) - F(x_0)}{x - x_0} = f(x_0) \implies F'(x) = f(x)
        \end{aligned}
        \]
    \end{Proof}

\subsection{Формула Ньютона-Лейбница}
    \begin{theorem}
        Если $f(x)$ непрерывна на $[\alpha; \beta]$ и $\Phi(x)$ первообразная $f(x)$ на $(\alpha; \beta)$, то\\
        \[
        \begin{aligned}
            &\forall a, b \in (\alpha; \beta)\\
            &\Int{a}{b}f(x)\diff x = \Phi(b) - \Phi(a)
        \end{aligned}
        \]
    \end{theorem}

    \begin{Proof}
        \[
        \begin{aligned}
            &F(x) = \Int{a}{x}f(t)\diff t \text{ - первообразная } f(x) \text{ на } (\alpha; \beta)\\
            &F(a) = 0\\
            &\exists c:\qquad F(x) = \Phi(x) + c \text{ на } (\alpha; \beta) \implies\\
        \end{aligned}
        \]
        \[
        \begin{aligned}
            \implies F(a) = \Phi(a) + c = 0 \implies c = -\Phi(a)\\
        \end{aligned}
        \]
        \[
        \begin{aligned}
            \Int{a}{b}f(x)\diff x = F(b) = \Phi(b) + c = \Phi(b) - \Phi(a)\\
            \text{Ч.Т.Д.}
        \end{aligned}
        \]
        
    \end{Proof}
