\section{Свойства определенного интеграла. Интеграл с переменным верхним пределом. Его свойства. Формула Ньютона-Лейбница.}

\subsection{Свойства определенного интеграла}
\begin{theorem}
    
    Если\\
    \[
    \begin{aligned}
        f(x) \in R[a; b]
    \end{aligned}
    \]
    То\\
   \[
        \forall \tau_n : d(\tau_n) \tounder{n \to \infty} 0,\quad \forall \xi_n:
        \quad\sigma(\tau_n; \xi_n) \tounder{n \to \infty} \Int{a}{b} f(x) \diff x = I
    \]
\end{theorem}
\begin{Proof}
    Есть:\\
    \[
        \forall \eps > 0~ \exists\delta(\eps)~ \forall\tau: ~d(\tau) < \delta(\eps),\quad
        \forall \xi:\quad |\sigma(\tau; \xi) - I| < \eps
    \]
    \[
        \forall\gamma > 0~\exists N(\gamma)~ \forall n > N(\gamma):~ d(\tau_n) < \gamma
    \]
    Нужно:\\
    \[
        \forall \eps > 0~ \exists N'(\eps)~ \forall n > N'(\eps):~ |\sigma(\tau_n; \xi_n) - I| < \eps\\
    \]

    Начиная с $N'(\eps) = N(\delta(\eps))$ всё выполнено
\end{Proof}

\begin{theorem}[Линейность]
    Если
    \[
        f(x), g(x) \in R[a; b]
    \]
    То
    \[
        \forall\alpha, \beta \in \R: \alpha f(x) + \beta g(x) \in R[a; b]
    \]
    И
    \[
        \Int{a}{b}{(\alpha f(x) + \beta g(x))\diff x} = \alpha\Int{a}{b}{f(x)\diff x} + \beta\Int{a}{b}{g(x)\diff x}
    \]
\end{theorem}

\begin{Proof}
\textbf{Интегрируемость:} 
$f,g$ ограничены $\Rightarrow \alpha f + \beta g$ ограничена.  
Множество точек разрыва:
\[
D(\alpha f + \beta g) \subseteq D(f) \cup D(g),
\]
а объединение множества меры ноль снова имеет меру ноль $\Rightarrow \alpha f + \beta g \in R[a,b]$.

\textbf{Свойство интеграла:} 
Для суммы Римана
\[
\sigma_{\alpha f + \beta g}(\tau;\xi) = \sum_{k=1}^n (\alpha f(\xi_k) + \beta g(\xi_k))\Delta x_k
= \alpha \sum_{k=1}^n f(\xi_k)\Delta x_k + \beta \sum_{k=1}^n g(\xi_k)\Delta x_k
= \alpha \sigma_f + \beta \sigma_g.
\]

Переходя к пределу $d(\tau)\to 0$, получаем:
\[
\int_a^b (\alpha f + \beta g) = \alpha \int_a^b f + \beta \int_a^b g.
\]
\end{Proof}
    
\begin{definition}
    $\boxed{2}$(Аддитивность)\\
    Если\\
    \[
        f(x) \in R[a; b]
    \]
    То\\
    \[
        \forall c \in (a; b): f(x) \in R[a; c] \text{ и } f(x) \in R[c; b]
    \]
    И\\
    \[
        \Int{a}{b}{f(x)\diff x} = \Int{a}{c}{f(x)\diff x} + \Int{c}{b}{f(x)\diff x}
    \]
\end{definition}
\begin{Proof}
    1) Доказательство интегрируемости очевидно\\
    2) Связь интегралов

    Построим последовательность $\tau'_n \text{ отрезка } [a; c] \text{ и возьмем } \xi'_n$\\
    Построим последовательность $\tau''_n \text{ отрезка } [c; b] \text{ и возьмем } \xi''_n$\\
    Объединим разбиения и получим $\tau_n \text{ отрезка } [a; b] \text{ и разбиение } \xi_n$\\
    \[
        \sigma_f(\tau_n; \xi_n) = \sigma_f(\tau'_n; \xi'_n) + \sigma_f(\tau''_n; \xi''_n)
    \]
    А значит
    \[
        \Int{a}{b}{f(x)\diff x} = \Int{a}{c}{f(x)\diff x} + \Int{c}{b}{f(x)\diff x}
    \]

\end{Proof}

\begin{definition}
    $\boxed{3}$\\
    3.1) Если $f(x) \in R[a; b]$, то $|f(x)| \in R[a; b]$\\
    3.2) Если $f(x), g(x) \in R[a; b]$ и $f(x) \leq g(x) \text{ на } [a; b]$, то $\Int{a}{b}{f(x)\diff x} \leq \Int{a}{b}{g(x)\diff x}$\\
    3.3) Если $f(x) \in R[a; b]$, то $\l|\Int{a}{b}{f(x)\diff x}\r| \leq \Int{a}{b}{|f(x)|\diff x}$ - Обобщенное неравенство треугольника\\
\end{definition}

\begin{Proof}[Докажем 3.3]
    \[
    \begin{aligned}
        &-|f(x)| \leq f(x) \leq |f(x)|\\
        &-\Int{a}{b}{|f(x)|\diff x} \leq \Int{a}{b}{f(x)\diff x} \leq \Int{a}{b}{|f(x)|\diff x}\\
    \end{aligned}
    \]
\end{Proof}

\begin{theorem}[$\boxed{4}$ Теорема о среднем]
    Если $f(x)$ - непрерывна на [a; b], то:\\
    \[
        \exists\xi\in(a; b):\quad \Int{a}{b}{f(x)\diff x} = f(\xi)(b - a)
    \]
\end{theorem}

\begin{Proof}
    Обозначим:\\
    \[
    \begin{aligned}
        &m = \inf f(x)\\
        &M = \sup f(x)\\
    \end{aligned} 
    \]
    Начало доказательства:\\
    \[
    \begin{aligned}
        m &\leq f(x) \leq M\\
        m(b - a) &\leq \Int{a}{b}{f(x)\diff x} \leq M(b - a)\\
        m &\leq \frac{1}{(b - a)}\Int{a}{b}{f(x)\diff x} \leq M\\
    \end{aligned}
    \]
    \[
    \begin{aligned}
        \exists\xi \in(a; b):\quad f(\xi) = \frac{1}{(b - a)}\Int{a}{b}{f(x)\diff x}
    \end{aligned}
    \]
\end{Proof}

\begin{remark}
    В предыдущем доказательстве равенство в неравенстве возможно только в случае, если $f(x) - const$
\end{remark}


\subsection{Интеграл с переменным верхним пределом}
\begin{definition}
    Пусть $f(x) \in R[\alpha; \beta],\quad a, x \in [\alpha; \beta]$\\
    Интегралом с переменным верхним пределом определим функцию $F(x) = \Int{a}{x}f(t)\diff t$
\end{definition}


\subsection{Свойства интеграла с переменным верхним пределом}
\begin{remark}
    \[
        F(a) = 0
    \]
\end{remark}

\begin{theorem}[Свойство 1 --- непрерывность]
    $F(x)$ непрерывна на $[\alpha; \beta]$
\end{theorem}

\begin{Proof}
    Пусть $x_0 \in [\alpha,\beta]$. Тогда
\[
F(x)-F(x_0)=\int_{x_0}^{x} f(t)\,dt.
\]

Так как $f$ интегрируема на $[\alpha,\beta]$, она ограничена, то есть существует $M>0$, такое что
\[
|f(t)| \le M \quad \text{для всех } t \in [\alpha,\beta].
\]

Тогда
\[
\left|F(x)-F(x_0)\right|
=
\left|\int_{x_0}^{x} f(t)\,dt\right|
\le
\int_{x_0}^{x} |f(t)|\,dt
\le
M|x-x_0|.
\]

Переходя к пределу при $x \to x_0$, получаем
\[
|F(x)-F(x_0)| \to 0,
\]
следовательно,
\[
\lim_{x \to x_0} F(x)=F(x_0).
\]

Значит, $F$ непрерывна в точке $x_0$. Так как $x_0$ произвольно, $F$ непрерывна на $[\alpha,\beta]$.
\end{Proof}

    \begin{theorem}
        Если $f(x)$ - непрерывна на $(\alpha; \beta)$, то $F(x)$ дифференцируема на $(\alpha; \beta)$ и $F'(x) = f(x)$\\
    \end{theorem}
\begin{Proof}
    Пусть $x_0\in (\alpha,\beta)$ и $\Delta x \neq 0$ малое. Тогда
\[
\frac{F(x_0+\Delta x)-F(x_0)}{\Delta x}
=
\frac{1}{\Delta x}\int_{x_0}^{x_0+\Delta x} f(t)\,dt.
\]

По теореме о среднем для интеграла существует $\xi \in (x_0, x_0+\Delta x)$, такое что
\[
\int_{x_0}^{x_0+\Delta x} f(t)\,dt = f(\xi)\,\Delta x.
\]

Следовательно,
\[
\frac{F(x_0+\Delta x)-F(x_0)}{\Delta x} = f(\xi).
\]

При $\Delta x \to 0$ имеем $\xi \to x_0$, а из непрерывности $f(\xi) \to f(x_0)$. Следовательно,
\[
F'(x_0) = \lim_{\Delta x \to 0} \frac{F(x_0+\Delta x)-F(x_0)}{\Delta x} = f(x_0).
\]
    \end{Proof}

\subsection{Формула Ньютона-Лейбница}
    \begin{theorem}
        Если $f(x)$ непрерывна на $[\alpha; \beta]$ и $\Phi(x)$ первообразная $f(x)$ на $(\alpha; \beta)$, то:
        \[
            \forall a, b \in (\alpha; \beta)
        \]
        \[
\Int{a}{b}f(x)\diff x = \Phi(b) - \Phi(a)
        \]
    \end{theorem}

    \begin{Proof}
Рассмотрим функцию с переменным верхним пределом:
\[
F(x) = \int_a^x f(t)\,dt.
\]

По теореме о производной интеграла имеем
\[
F'(x) = f(x) \quad \text{для всех } x \in (\alpha,\beta),
\]
то есть $F$ и $\Phi$ — первообразные одной и той же функции. Значит, они отличаются константой:
\[
F(x) = \Phi(x) + C.
\]

Подставим $x=a$:
\[
F(a) = \int_a^a f(t)\,dt = 0 = \Phi(a) + C \quad \Rightarrow \quad C = -\Phi(a).
\]

Следовательно,
\[
F(b) = \int_a^b f(t)\,dt = \Phi(b) + C = \Phi(b) - \Phi(a),
\]
что и требовалось доказать.
    \end{Proof}
