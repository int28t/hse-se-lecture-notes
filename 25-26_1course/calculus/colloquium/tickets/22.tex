\section{Верхние и нижние суммы Дарбу. Их свойства. Верхний и нижний интеграл Дарбу. Критерий интегрируемости Дарбу.}

\subsection{Верхние и нижние суммы Дарбу}
\begin{definition}
    Верхняя сумма Дарбу: $S_\tau = \Sum{i = 1}{n}M_i\cdot|\D x_i|$\\
    $M_i = \subs{sup f(x)}{x \in \D x_i}$
\end{definition}

\begin{definition}
    Нижняя сумма Дарбу: $s_\tau = \Sum{i = 1}{n}m_i\cdot|x_i|$\\
    $m_i = \subs{inf f(x)}{x \in \D x_i}$
\end{definition}

\subsection{Свойства сумм Дарбу}
\begin{definition}
    Свойства сумм Дарбу:\\
    1. $S_\tau, s_\tau$ - Определены (Т.к. интегрируются только огр функции)\\
    2. $s_\tau \leq S_\tau$ при одинаковых разбиениях\\
    3. $S_\tau = \subs{sup~\sigma(\tau, \xi)}{\xi}$\\
    $s_\tau = \subs{inf~\sigma(\tau, \xi)}{\xi}$\\
    4. $\text{Разбиение } \tau' = \Subst{\{[x_{i-1}'; x_i']\}}{i = 1}{n}$ является измельчением $\tau = \Subst{\{[x_{j-1}; x_j]\}}{j = 1}{k}$\\
    $\forall j ~\exists i: x_j = x_i' (n \geq k)$\\
    Обозначается: $\tau' \succ \tau$ 
\end{definition}

\begin{theorem}[4 свойство]
    Дано: $\tau' \succ \tau$\\
    $S_{\tau'} \leq S_\tau$\\
    $s_{\tau'} \geq s_\tau$
\end{theorem}

\begin{Proof}
    1. Пусть $\tau' \succ \tau \text{ причем } \tau' \text{ отличается от } \tau \text{добавлением одной точки в } \D x_i$\\

    $\text{Рассмотрим разность } S_\tau $ и $S_{\tau'}$\\
    Наша новая точка на разбиении находиться на позиции *\\
    $S_\tau - S_{\tau'} = \Sum{i = 1}{n}M_i\cdot|\D x_i| - \Sum{i = 1}{n + 1}M'_i\cdot|\D x'_i| = M_i\cdot|\D x_i| - (M'_i\cdot|\D x'_i| + M'_i\cdot|\D x'_{i+1}|) = $\\
    Последнее равенство выполнено, потому все подотрезки $\D x_i$ совпадают кроме позиций i и i + 1\\
    $|\D x_i| = |\D x'_i| + |\D x'_{i + 1}|$\\
    $ = (M_i - M'_i)\cdot|\D x'_i| + (M_i - M'_{i + 1})\cdot|\D x'_{i + 1}|$\\
    Разность между $M_i \text{ и } M'_i\geq 0$ и разность между $M_i \text{ и } M'_{i + 1} \geq 0$ \\
    Сумма двух неотрицательных чисел всегда положительна или равна 0

    2. $\tau' \succ \tau$\\
    Построим последовательность $\tau^*_k$\\
    $\tau^*_k = \tau' \succ \tau^*_{k - 1} ... \succ \tau^*_1 = \tau$\\
    $\tau^*_{i} \succ \tau^*_{k - 1}$ отличаются на одну точку $\implies S_{\tau^*_{i - 1}} \geq S_{\tau^*_{i}}$\\
    Из такого ряда неравенств получим, что $S_{\tau'} \geq S_{\tau}$\\
    Аналогично доказывается для нижних сумм Дарбу.
\end{Proof}

\begin{theorem}[5 свойство]
    $\forall \tau_1, \tau_2: s_{\tau_1} \leq S_{\tau_2}$\\
\end{theorem}

\begin{Proof}
    Построим измельчение $\tau$, которое состоит из точек $\tau_1$ и $\tau_2 \implies$\\
    $\implies \tau \succ \tau_1 \text{ и } \tau \succ \tau_2$\\
    Тогда: $s_{\tau_1} \leq s_\tau \leq S_\tau \leq S_{\tau_2}$ чтд
\end{Proof}

\subsection{Верхний и нижний интеграл Дарбу}

\begin{definition}
    Верхний интеграл Дарбу: $I^* = \subs{inf(S_\tau)}{\tau}$

\end{definition}

\begin{definition}
    Нижний интеграл Дарбу: $I_* = \subs{sup(s_\tau)}{\tau}$

\end{definition}

\subsection{Критерий интегрируемости Дарбу}

\begin{theorem}[Критерий интегрируемости Дарбу]
    \[
    f(x) \in R[a, b] \iff 
    \begin{cases} 
        f(x) \text{ - огр на } [a; b]\\
        I^* = I_*
    \end{cases}
    \]
\end{theorem}


\begin{Proof}[Достаточность]

    Если $I^* \neq I_*$, то $I^* > I_*$ (Доказывается от противного -> смотрим на рисунок -> видим противоречие с 5 свойством сумм Дарбу)\\

$\forall\eps > 0~\exists\delta > 0~\forall\tau~d(\tau)~ <~\delta~ \forall\xi:~|\sigma(\tau;\xi) - I| < \eps$\\

Возьмем $\eps = \frac{\gamma}{3}$ и перейдем к рисунку и увидим противоречие.\\

\begin{Proof}[Необходимость]
    Докажем две леммы, которые доказывают критерий Дарбу:\\
    1. $\tau' \succ \tau$ отличаются p точками $p \in \N \implies S_\tau - S_{\tau'} \leq (M - m)\cdot p\cdot\delta$\\ $d(\tau) < \delta$ \\
    p = 1:\\
    $S_\tau - S_{\tau'} = M_i|\D x_i| - (M'_i|\D x'_i| + M'_{i+1}|\D x'_{i+1}| = (M_i - M'_i)|\D x'_i| + (M_i - M'_{i+1})|\D x'_{i+1}|) \leq (M - m)|\D x'_{i}| + (M - m)|\D x'_{i+1}| = (M - m)|\D x_{i}| < (M - m)\cdot\delta$\\

    2. Лемма Дарбу: $I^* = \Lim{d(\tau)}{0}S_\tau \iff \forall\eps > 0~\exists\delta~\forall\tau~d(\tau) < \delta: |i^* - S_\tau| < \eps$\\
    Шаг 1: если $M = m = const = f(x)$ на [a; b] очевидно, что $\forall\tau ~ \forall\xi:~\sigma(\tau, \xi) = const\cdot[a; b]$\\
