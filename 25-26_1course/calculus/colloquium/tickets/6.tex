\begin{definition}
	\textbf{Функцией} $f$ отображающей множество $X$ во множество $Y$ $(f : X \to Y)$ называется множество упорядоченных пар $\{(x, y)\}_{x \in X, y \in Y}$, причем каждый элемент $X$ встречается в какой-то паре на первой позиции и только один раз.

	В частности, $X$ называется областью определения функции и обозначается $D_f$

	Множество вторых элементов пар обозначается $E_f$, множество значений функции. $E_f \subset Y$
\end{definition}

\begin{definition}
	\textbf{График функции} --- множество точек на декартовой (координатной) плоскости, координаты которых --- пары функции
\end{definition}
\subsection{Инъекция, сюрьекция, биекция} 
\begin{definition}
	Функция называется \textbf{инъекцией}, если вторые элементы пар не повторяются
\end{definition}
\begin{definition}
	Функция называется \textbf{сюрьекцией}, если $E_f = Y$
	\[
		\forall y \in Y\ \exists x \in X\ : f(x) = y
	\]
\end{definition}
\begin{definition}
	Функция называется \textbf{биекцией}, если она инъекция и сюрьекция
\end{definition}
\subsection{Обратимость функции}
\begin{definition}
	Функция $f^{-1} : Y \to X$ называется \textbf{обратной} к $f : X \to Y$, если пары $f^{-1}$ -- пары $f$, где элементы поменяны местами 
\end{definition}

\begin{remark}
	Чтобы $f : X \to Y$ была обратима нужно, чтобы она была биекцией
\end{remark}

\begin{example}
	\[
		f(x) = \sin(x),\ \R \to \R\ \text{--- не обратима}
	\]
	\[
		f(x) = \sin(x),\ \l[-\frac{\pi}{2};\frac{\pi}{2} \r] \to [-1; 1]
	\]
	Тогда обратима
	\[
		f^{-1}(x) = \arcsin (y) \in \l[-\frac{\pi}{2};\frac{\pi}{2} \r]
	\]
\end{example}

\subsection{Теорема о достаточном условии обратимости}
\begin{theorem}[Теорема о достаточном условии обратимости]
	Если $f(x) $ --- строго монотонна на $X$, то $f(x)$ обратима
\end{theorem}

\begin{Proof}
Предположим, что она не обратима (а значит, она не инъективна)

Это значит что:
$\exists  x_1 \neq x_2: y_1 = y_2$

Предположим, что $f(x)$ возрастает

Тогда $x_1 < x_2 \Rightarrow y_1 < y_2$
и $x_1 > x_2 \Rightarrow y_1 > y_2$

Оба утверждения противоречат тому что $y_1 = y_2$
\end{Proof}

\subsection{Критерий обратимости непрерывной на отрезке функции}
\begin{theorem}[Теорема о критерии обратимости непрерывной на отрезке функции]
	Если $f(x) $ непрерывна на $[a;b]$, то:
	\[
		f(x) \text{ обратима на } [a; b] \Leftrightarrow f(x) \text{ строго монотонна на } [a; b]
	\]
\end{theorem}

\begin{Proof}
	\fbox{$\Leftarrow$} Доказано раннее (см. "Теорема о достаточном условии обратимости")

	Если функция строго монотонная на каком-то множестве то она обратима на этом множестве

\fbox{$\Rightarrow$} Предположим противное (Сформулируем что значит что "функция не (строго монотонна)")

1. $\exists x_1 \neq x_2: y_1 \neq y_2$

(i.e. если где-то нарушается строгость то это сразу приводит к противоречию)

2. $x_1 < x_2 < x_3: (\underbrace{y_1 < y_2 > y_3}_{\cap} \lor \underbrace{y_1 > y_2 < y_3}_{\cup})$

Без ограничения общности рассмотрим только случай $x_1 < x_2 < x_3: y_1 < y_2 > y_3$

Хотим прийти к противоречие с обратимостью (i.e. с инъективностью)
а значит найти $x_i\neq x_j:y_i=y_j$

\end{Proof}

	\end{cases}
\]

Поскольку $z_1$ и $z_2$ принадлежат разным интервалам то $z_1 \neq z_2$

Но $f(z_1) = f(z_2)$ --- противоречие


\end{Proof}

\begin{remark}[Словарик]
	Без ограничений общности --- кажется что мы накладываем ограничение превращая в частный случай, но на самом деле не влияя на справедливость доказательства в целом, так как остальные случаи рассматриваются похожим образом
\end{remark}


\subsection{Теорема о непрерывности и монотонности обратной функции на отрезке}
\begin{theorem}[Теорема о непрерывности и монотонности обратной функции на отрезке]
	$$f(x)$$
	\begin{enumerate}
		\item[1)] $f(x)$ определена на $[a;b]=D_f$ 
		\item[2)] $f(x)$ непрерывна на $[a;b]$
		\item[3)] $f(x)$ строго монотонна на $[a;b]$, то
	\end{enumerate}
	$$\exists f^{-1}(y)$$
	\begin{enumerate}
		\item[1)] область определения --- отрезок с концами $f(a)$ и $f(b)$
		\item[2)] непрерывна на этом отрезке 
		\item[3)] $\underbrace{\text{также}}_{\text{=в том же направлении}}$ строго монотонна на этом отрезке
	\end{enumerate}
\end{theorem}
\begin{Proof}
	\begin{enumerate}
		\item[\bf 1)] Следствие из теоремы о промежуточных значениях
		(если функция непрерывна на отрезке то множество её значений есть отрезок от инфинума до супремума)
		\item[\bf3)] Без ограничения общности: пусть $f(x)$ возрастает

		Предположим противное: $f^{-1}$ не возрастает: (построим отрицание)

		$\exists y_1, y_2$ принадлежащие отрезку $: (y_1 < y_2) \land x_1 \geq x_2$

		$x_1 \geq x_2 \Rightarrow \underbrace{f(x_1)}_{y_1} \geq \underbrace{f(x_2)}_{y_2}$


		т.е. $\exists y_1, y_2$ принадлежащие отрезку $: (y_1 < y_2) \land y_1 \geq y_2$

		противоречие

		\item[\bf 2)] Нужно доказать, что функция непрерывна на отрезке (она непрерывна во всех точках а на концах односторонние пределы)

		Докажем для внутренних точек (для концов делается также только надо будет смотреть на полуокрестности)

		Рассмотрим: $y_0 \in (f(a); f(b))$

		Всё ещё полагаем какую-либо монотонность. Хотим:

		\[
			\Lim{y}{y_0} f^{-1}(y) = f^{-1}(y_0)  
		\]
		\[
			\forall \eps > 0\ \exists \delta = \delta (\eps )\ \forall y \in U_{\delta }(y_0)\ \underbrace{|f^{-1}(y) - \underbrace{f^{-1}(y_0)}_{x_0 \in (a;b)}| < \eps }_{\Leftrightarrow f^{-1}(y)~\in~U_{\epsilon}(x_0)}
		\]

		Без ограничений общности считаем, что:
		\[
			U_{\eps }(x_0) \subset (a;b) 
		\]
		% --- вставить сюда (начало TODO) ---
Возьмём произвольное $\varepsilon>0$. Так как $x_0=f^{-1}(y_0)\in(a,b)$, можно выбирать $\varepsilon$ настолько малым, чтобы
\[
U_{\varepsilon}(x_0)=(x_0-\varepsilon,x_0+\varepsilon)\subset (a,b).
\]
Положим
\[
y_1:=f(x_0-\varepsilon),\qquad y_2:=f(x_0+\varepsilon).
\]
Поскольку $f$ возрастает, имеем $y_1<y_0<y_2$. Следовательно числа
\[
y_0-y_1>0,\qquad y_2-y_0>0
\]
и можно положить
\[
\delta:=\min\{\,y_0-y_1,\;y_2-y_0\,\}>0.
\]

Покажем, что при любом $y$ с $|y-y_0|<\delta$ выполняется $|f^{-1}(y)-x_0|<\varepsilon$.
Если $|y-y_0|<\delta$, то $y_1<y<y_2$ (так как $\delta\le y_0-y_1$ и $\delta\le y_2-y_0$). Применяя монотонность $f^{-1}$ получаем
\[
f^{-1}(y_1)=x_0-\varepsilon < f^{-1}(y) < f^{-1}(y_2)=x_0+\varepsilon.
\]
Отсюда $|f^{-1}(y)-x_0|<\varepsilon$. Это доказывает, что
\[
\lim_{y\to y_0} f^{-1}(y)=f^{-1}(y_0),
\]
то есть $f^{-1}$ непрерывна в точке $y_0$.

(Если $f$ убывает, в рассуждении меняются местами неравенства, но схема остаётся та же.) 
% --- вставить сюда (конец TODO) ---

	\end{enumerate}

\end{Proof}

\begin{example}[Следствие]
	Промежутками называются все интервалы (когда концы в том числе бесконечные)

	$[a;b]$, где $a,b \in \R$

	полуинтервалы:

	$(a,b], [b, a]$, где $a \in \overline{\R}$, $b \in \R$


	\textbf{Следствие} Если $f(x)$ определена, непрерывна и строго монотонна на промежутке $(a; b)$ то $f^{-1}$ определена, непрерывна и строго монотонна на $E_f$

	Без доказательства
\end{example}

\subsection{Непрерывности корня}
\begin{theorem}[Непрерывность корня]
	$y = x^n, n \in \N$

	если $n$ нечётно то $D_f = \R$ возрастает, непрерывно, $E_f = \R$

	если $n$ чётно то $D_f = [0; +\infty)$ возрастает, непрерывно, $E_f = [0; +\infty)$

	(будем ограничивать функцию для монотонности)

	$y = \sqrt[n]x$ для нечётных $n, D_f = \R$ возрастает, непрерывнo

	для чётных $n, D_f = [0; +\infty) $ возрастает, непрерывнo

\end{theorem}
