\section{Формула подстановки и формула замены переменной в неопределенном интеграле + теорема, на которой они основаны. Формула интегрирования по частям.}

\begin{definition}
    Формула подстановки: $\int f(\vphi(t)) \cdot \vphi'(x) \diff t = \subs{\int f(x) \diff x} {x = \vphi t}$ 

    Формула замены переменной в неопределенном интеграле:
    
    $\int f(x) \diff x \eqind{x = \vphi(t)} \subs{\int f(\vphi(t)) \cdot \vphi'(t)}{t = \vphi^{-1}(x)} = G(t) + C = G(\vphi^{-1}(x)) + C;~C \in \R$
\end{definition}

\begin{theorem}
    Если у $f(x) ~\exists$ первообразная на интервале (а, b) и $\vphi(t)$ диффeренцруема на (c, d) и $\subs{\vphi(t)}{t \in (c,~d)} \implies \int f(\vphi(t))\cdot\vphi'(t)\diff t = F(\vphi(t)) + C;~C\in\R$
    \begin{proof}
    Продифференцируем правую часть:

    $F(\vphi(t)) + C = F'(\vphi(t))\cdot\vphi'(t) = f(\vphi(t))\cdot\vphi'(t)$
    \end{proof}
\end{theorem}
    
\begin{theorem}
    Интегрирование по частям: 

    $f(x), g(x) -~$ дифференцируемы на (a, b) $\implies \int f(x)\diff g(x) = f(x)\cdot g(x) - \int g(x)\diff f(x)$

    \begin{proof}
        Проинтегрируем $(f(x)\cdot g(x))'$

        $(f(x)\cdot g(x))' = f'(x)g(x) + f(x)g'(x)$\\
        $\int (f'(x)g(x) + f(x)g'(x))\diff x = f(x)\cdot g(x) + C;~ C \in \R$\\
        $\int (f'(x)g(x) + f(x)g'(x))\diff x = \int g(x)\diff f(x) + \int f(x)\diff g(x)$\\
        $\int f(x)\diff g(x) = f(x)\cdot g(x) - \int g(x) \diff f(x)$

    \end{proof}

\end{theorem}
