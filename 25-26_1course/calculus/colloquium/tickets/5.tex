\section{Свойства непрерывных на отрезке функций:
достижение точной верхней и нижней грани.
Теорема о промежуточных значениях.}

\begin{definition}
    Функция непрерывна на $[a; b]$, если

    \begin{itemize}
        \item[1)] $[a;b] \subset D_f$
        \item[2)] $f(x)$ непрерывна $ \forall x_0 \in (a;b)$
        \item[3)] $\Lim{x}{a^+} f(x) = f(a)\ \Lim{x}{b^-} f(x) = f(b)$
    \end{itemize}
\end{definition}

\begin{remark}
    В дальнейшем будем считать что $D_f = [a;b]$
\end{remark}

\subsection{Ограниченность непрерывной функции на отрезке, достижение точной верхней и нижней грани}

\begin{theorem}
    Если функция непрерывна на отрезке $[a;b]$, то 
    \begin{itemize}
        \item[1)] она ограничена на $[a;b]$
        \item[2)] достигает $\underset{x \in [a;b]}{\sup f(x)} = \sup E_f$ и $\underset{x \in [a;b]}{\inf f(x)} = \inf E_f$
    \end{itemize}
\end{theorem}

\begin{Proof}
    Пусть $M = \underset{x \in [a;b]}{\sup f(x)} \in \overline{\R}, m = \underset{x \in [a;b]}{\inf f(x)} \in \overline{\R}$

    Хотим (докажем сверху):
    \begin{itemize}
        \item[1)] $M \in \R$
        \item[2)] $\exists x_0 \in [a, b] : f(x_0) = M$ 
    \end{itemize}

    Докажем 2), 1) следует из него (очевидно, что если $\exists x_0 : f(x_0) = M$, то $M \neq \infty$)

    Построим последовательность, которая возрастает к числу $M$: $a_n \uparrow M : a_n$ строго возрастает и $\Lim{n}{\infty} a_n = M$. Тогда: 
    \[
        \forall n\ \exists x_n \in [a;b]\ f(x_n) \geq a_n
    \]
\end{Proof}

\begin{Proof}[]
Предположим противное:
\[
    \exists n_0\ \forall x \in [a;b]\ f(x) < a_{n_0}
\]
$a_{n_0}$ --- верхняя грань для $E_f$, $\bot$

Построили $\{x_n\}$: 
\[
    \underbrace{a_n}_{\to M} \leq f(x_n) \leq \underbrace{M}_{\to M}
\]
\[
    \Lim{n}{\infty} f(x_n) = M
\]

$x_n \in [a;b] \text{ --- ограничена. Тогда по теореме Б-В } \Rightarrow \exists x_{n_k} : x_{n_k} \underset{k \to \infty}{\rightarrow} x_0$

Так как $x_{n_k} \in [a;b]$ и отрезок $[a;b]$ является замкнутым, то предел последовательности также принадлежит этому отрезку, то есть
\[
x_0 \in [a;b].
\]

\[
    f(x_{n_k}) \xrightarrow[k \to \infty ]{} M
\]
Определение непрерывности по Гейне:
\[
    f(x_{n_k}) \xrightarrow[k \to \infty ]{} f(x_0)
\]
По теореме о единственности предела:
\[
    M = f(x_0)
\]
\end{Proof}

\begin{remark}
То есть точные верхняя и нижняя грани совпадают соответственно
с максимумом и минимумом функции и достигаются в некоторых точках.
\end{remark}

\subsection{Теорема о промежуточных значениях}

\begin{theorem}
    Непрерывная на отрезке функция принимает все промежуточные значения, т.е $f(x), [a;b]$

    $$A = f(x_1),\ B = f(x_2),\ A < B,\ x_1, x_2 \in [a;b]$$

    тогда:
    $$\forall C \in (A;B)\ \exists x_0 \in [a;b] : f(x_0) = C$$
\end{theorem}

{\bf Следствие} $f(x)\ \text{непрерывна на}\ [a;b] = D_f, \text{то}\ E_f = [m; M]$

{\bf Следствие.} Если $f$ непрерывна на $[a,b]$ и
\[
f(a)\cdot f(b)<0,
\]
то существует $c\in(a,b)$ такое, что
\[
f(c)=0.
\]

\begin{Proof}
Построим последовательность вложенных отрезков $\l\{[a_k; b_k] \r\}_{k \in \N}$, будем считать $x_1 < x_2$

$[a_1; b_1] = [x_1; x_2]$

\[
    f(a_1) = f(x_1) < C,\ f(b_1) = f(x_2) > C
\]

Возьмем середину $x_3 = \frac{x1 + x2}{2}$

\begin{itemize}
    \item[1.] $f(x_3) = C$ ч.т.д.
    \item[2.] $f(x_3) < C$ $[a_2; b_2] = [x_3; b_1]$
    \item[3.] $f(x_3) > C$ $[a_2; b_2] = [a_1; x_3]$
\end{itemize}

По теореме Вейерштрасса:

$\{a_k\}_k$ --- неубывающая и огр сверху b $\Rightarrow a_k \xrightarrow[k \to \infty ]{} \alpha $

$\{b_k\}_k$ --- невозрастает и огр снизу a $\Rightarrow b_k \xrightarrow[k \to \infty ]{} \beta $

$\underbrace{b_k}_{\to \beta } - \underbrace{a_k}_{\to \alpha } = \frac{1}{2^{k - 1}} (x_2 - x_1) \underset{k \to \infty}{\to} 0 \Rightarrow \alpha = \beta $
\[
    \underbrace{f(a_k)}_{\xrightarrow[k \to \infty ]{} f(\alpha) } < C
\]
По теореме о предельном переходе:
\[
    f(\alpha) \leq C
\]
Аналогично $f(b_k)$

Тогда если $\alpha = \beta = x_0$:
\[
    f(x_0) = f(\alpha) = f(\beta) = C
\]
\end{Proof}

\subsection{Примеры}

\begin{example}
Рассмотрим функцию
\[
f(x)=x^3-x
\]
на отрезке $[-2,2]$.

Она непрерывна (многочлен), следовательно:
\begin{itemize}
\item ограничена;
\item достигает максимума и минимума;
\item принимает все значения между $f(-2)$ и $f(2)$.
\end{itemize}
\end{example}

\begin{example}
Докажем, что уравнение
\[
x^3+x-1=0
\]
имеет корень на $(0,1)$.

Пусть
\[
f(x)=x^3+x-1.
\]
Тогда
\[
f(0)=-1<0,
\qquad
f(1)=1>0.
\]
Функция непрерывна, значит существует
$c\in(0,1)$ такое, что $f(c)=0$.
\end{example}
