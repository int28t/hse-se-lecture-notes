\section{Понятия разбиения отрезка, разметки разбиения, интегральной суммы. Определенный интеграл Римана на отрезке. Ограниченность интегрируемой функции.}


\subsection{Понятия разбиения отрезка, разметки разбиения, интегральной суммы.}
\begin{definition}[Разбиение отрезка]
    Разбиение отрезка: $\tau = \Subst{\{[x_{i-1}, x_i]\}}{i = 1}{n}$

    $a = x_0 < x_1 < ... < x_n = b$
\end{definition}

\begin{definition}[Диаметр разбиения]
    Диаметр разбиения - макс расстояние между $x_i \text{ и } x_{i+1}$\\

    $d(\tau) = \Subst{max\{|{\D x_i}|\}}{i = 1}{n}$
\end{definition}

\begin{definition}[Разметка разбиения]
    Разметка разбиения $\xi = \xi(\tau) = \Subst{\{\xi_i\}}{i = 1}{n}$ - множество значений, попавших в $\D x_i$
\end{definition}

\begin{definition}[Интегральная сумма]
    Интегральная сумма: $\sigma (\tau, \xi) = \Sum{i = 1}{n}f(\xi_i)|\D x_i|$
\end{definition}

\subsection{Определённый интеграл Римана на отрезке.}
\begin{definition}[Определённый интеграл Римана на отрезке. Определение по Коши.]
    
    \noindent
\[
\forall \varepsilon>0\ \exists \delta>0\ \forall \tau:
\quad d(\tau)<\delta,\ \forall \xi \quad |\sigma(\tau,\xi)-I|<\varepsilon
\]
\[
f(x)\in R[a,b]
\quad
\text{(интегрируется по Риману на отрезке)}
\]
\[
I = \int\limits_a^b f(x)\,dx
\]
\end{definition}


\subsection{Ограниченность интегрируемой функции}
\begin{theorem}[Теорема об ограниченности интегрируемой функции]
    \noindent
\[
f(x)\in R[a,b]\;\Rightarrow\; f(x)\ \text{--- огр на }[a,b].
\]
\end{theorem}

\begin{Proof}
\noindent
Предположим противное: $f(x)$ - неограниченная.\quad $\varepsilon_0=\dfrac12$.

\[
\exists \delta_0>0\ \forall \tau:
\quad d(\tau)<\delta_0,\ \forall \xi \quad |\sigma(\tau,\xi)-I|<\frac12
\]

\noindent
Зафиксируем $\tau_{0}$.

\[
\exists \Delta x_i^0:\ f(x)\ \text{не огр на}\ \Delta x_i^0.
\]
Пусть $i=1$.

\noindent
Зафиксируем $\xi_2^0,\dots,\xi_n^0$.

\[
\left|
\underbrace{f(\xi_1)}_{\text{сколь угодно большое}}\!|\Delta x_1^0|
+\underbrace{\sum_{i=2}^{n} f(\xi_i^0)\,|\Delta x_i^0|-I}_{\text{const}}
\right|
<\frac12\quad
\forall \xi_1 \in \Delta x_1^{0}
\]

\noindent
По неравенству треугольника получаем:
\[
|f(\xi_1)|\,|\Delta x_1^0|
<
\frac12
+
\left|
\sum_{i=2}^{n} f(\xi_i^0)\,|\Delta x_i^0|
-I
\right| \quad
\forall \xi_1 \in \Delta x_1^{0}
\] - $\bot$




\end{Proof}
